302  private void startApp(final boolean firstStart)
303  {
304  try
305  {
306  ActivityInfo ai=getPackageManager().getActivityInfo(getComponentName(), PackageManager.GET_META_DATA);
307  if (ai.metaData.containsKey("android.app.qt_libs_resource_id"))
308  {
309  int resourceId = ai.metaData.getInt("android.app.qt_libs_resource_id");
310  m_qtLibs=getResources().getStringArray(resourceId);
311  }
312  if (getIntent().getExtras()!= null && getIntent().getExtras().containsKey("use_local_qt_libs")
313  && getIntent().getExtras().getString("use_local_qt_libs").equals("true"))
314  {
315  ArrayList<String> libraryList= new ArrayList<String>();
316 
317  String localPrefix="/data/local/qt/";
318  if (getIntent().getExtras().containsKey("libs_prefix"))
319  localPrefix=getIntent().getExtras().getString("libs_prefix");
320 
321  if (m_qtLibs != null)
322  for(int i=0;i<m_qtLibs.length;i++)
323  {
324  libraryList.add(localPrefix+"lib/lib"+m_qtLibs[i]+".so");
325  }
326 
327  if (getIntent().getExtras().containsKey("load_local_libs"))
328  {
329  String []extraLibs=getIntent().getExtras().getString("load_local_libs").split(":");
330  for (String lib:extraLibs)
331  if (lib.length()>0)
332  libraryList.add(localPrefix+lib);
333  }
334 
335  String dexPaths = new String();
336  String pathSeparator = System.getProperty("path.separator", ":");
337  if (getIntent().getExtras().containsKey("load_local_jars"))
338  {
339  String []jarFiles=getIntent().getExtras().getString("load_local_jars").split(":");
340  for (String jar:jarFiles)
341  if (jar.length()>0)
342  {
343  if (dexPaths.length()>0)
344  dexPaths+=pathSeparator;
345  dexPaths+=localPrefix+jar;
346  }
347  }
348 
349  Bundle loaderParams = new Bundle();
350  loaderParams.putInt(ERROR_CODE_KEY, 0);
351  loaderParams.putString(DEX_PATH_KEY, dexPaths);
352  loaderParams.putString(LOADER_CLASS_NAME_KEY, getIntent().getExtras().containsKey("loader_class_name")
353  ?getIntent().getExtras().getString("loader_class_name")
354  :"org.kde.necessitas.industrius.QtActivityDelegate");
355  loaderParams.putStringArrayList(NATIVE_LIBRARIES_KEY, libraryList);
356  loaderParams.putString(ENVIRONMENT_VARIABLES_KEY,"QML_IMPORT_PATH="+localPrefix+"/imports\tQT_PLUGIN_PATH="+localPrefix+"/plugins");
357  loaderParams.putString(APPLICATION_PARAMETERS_KEY,"-platform\tandroid");
358  loadApplication(loaderParams);
359  return;
360  }
361 
362  try {
363  if (!bindService(new Intent(org.kde.necessitas.ministro.IMinistro.class.getCanonicalName()), m_ministroConnection, Context.BIND_AUTO_CREATE))
364  throw new SecurityException("");
365  } catch (Exception e) {
366  if (firstStart)
367  {
368  String msg="This application requires Ministro service. Would you like to install it?";
369  if (m_activityInfo != null && m_activityInfo.metaData.containsKey("android.app.ministro_needed_msg"))
370  msg=m_activityInfo.metaData.getString("android.app.ministro_needed_msg");
371  downloadUpgradeMinistro(msg);
372  }
373  else
374  {
375  ministroNotFound();
376  }
377  }
378  }
379  catch (Exception e)
380  {
381  Log.e(QtApplication.QtTAG, "Can't create main activity", e);
382  }
383  }
